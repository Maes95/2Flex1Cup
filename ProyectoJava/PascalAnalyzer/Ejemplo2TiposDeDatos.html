type
TVector = array [ 25 .. 25 ] of REAL;
TPunto = record
coordenadaX : INTEGER;
coordenadaY : INTEGER;
end;
var
valor1 : TVector;
valor2 : TPunto;
begin
valor1[ 0 ] := 1;
valor2.coordenadaX := 1;
valor2.coordenadaY := 2;
end.
