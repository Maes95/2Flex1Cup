function areaCuadrado ( lado: REAL): REAL;
type
miArray = array [ 10 .. 5 ] of INTEGER;
var
resultado : REAL;
unArray : miArray;
otroArray : miArray;
begin
resultado := +0.0;
unArray[ 1 ] := 0;
resultado := lado * lado + miArray[1 + 2];
unArray := otroArray;
areaCuadrado := resultado;
end;
